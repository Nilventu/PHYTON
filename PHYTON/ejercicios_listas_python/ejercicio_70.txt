# 70. Crea un programa que permita introducir x palabras en una lista llamada lista1. Una vezintroducidas, crea una nueva lista, llamada lista2, exactamente igual a #lista1. Se deben mostrar porpantalla el contenidos de lista1 en orden ascendente y lista2 en orden descendente. Respeta elformato de entrada y salida tal y como se #muestra en el testeo.

lista1 = []
cantidad = int(input("Introduce la cantidad de palabras: "))
for i in range(cantidad):
    palabra = input(f"Introduce la palabra {i+1}: ")
    lista1.append(palabra)

lista2 = []
for palabra in lista1:
    lista2.append(palabra)

for i in range(len(lista1)):
    for j in range(i + 1, len(lista1)):
        if lista1[i] > lista1[j]:
  
            lista1[i], lista1[j] = lista1[j], lista1[i]

for i in range(len(lista2)):
    for j in range(i + 1, len(lista2)):
        if lista2[i] < lista2[j]:
    
            lista2[i], lista2[j] = lista2[j], lista2[i]


print("Lista ordenada ascendente:", lista1)
print("Lista ordenada descendente:", lista2)
